1. e.preventDefault()

function sendMessage(e) {
    e.preventDefault()  // Submitting form would refresh the page, so we prevent page reload
    const input = document.querySelector('input')
    if (input.value) {
        ws.send(input.value)
        input.value = ""
    }
    input.focus()  // Focus the input after sending the message
}

document.querySelector('form').addEventListener('submit', sendMessage)

Usually used to prevent page reload when submit form.

handle close()

// credit to: https://gist.github.com/benbjohnson/9eebd201ec096ab6430e1f33411e6427
func doSomething() error {
	f, err := os.Create("foo")
	if err != nil {
		return err
	}
  // ensure that it's definitely closed by the end
	defer f.Close()
	if _, err := f.Write([]byte("bar"); err != nil {
		return err
	}
  return f.Close()
}

Why we bother to care about getting error from f.Close()? We have got error from f.Write(). Well, there are a lot to say, learn more: Don’t Defer Close() on Writable Files - Go Notes
